\begin{frame}[fragile]
  \frametitle{Objetivos}
  
  \begin{block}{Objetivo General}
    Desarrollar una PINN basada en ecuaciones de Navier-Stokes y registros de estaciones meteorológicas para estimar presión y componentes del viento sobre una malla espacio-temporal en el litoral Caribe.
  \end{block}
  
  \begin{block}{Objetivos Específicos}
    \begin{enumerate}
      \item Organizar y preprocesar datos de estaciones del IDEAM (diciembre 2025)
      \item Diseñar una PINN que incorpore restricciones de N-S en su función de pérdida
      \item Evaluar desempeño mediante métricas estándar (RMSE, MAE, $R^2$) en datos de prueba
    \end{enumerate}
  \end{block}
  
  \note[item]{Tiempo estimado: 2-3 minutos}
\end{frame}
